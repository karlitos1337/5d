def hebbian_plasticity(w, s_pre, s_post, learning_rate=0.01):
    """Neurons that fire together, wire together"""
    dw = learning_rate * outer(s_post, s_pre)
    w_new = w + dw
    w_new = clip(w_new, 0, 1)  # keep in valid range
